[[value.fromWord32]]
== fromWord32

[source.no-repl,motoko]
----
let fromWord32 : (w : Word32) -> Char
----

Convert word `w` to a character.
Traps if `w` is not a valid Unicode scalar value.
Value `w` is valid if, and only if, `w < 0xD800 or (0xE000 <= w and w <= 0x10FFFF)`.
